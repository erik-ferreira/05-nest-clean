# Anotações

## Docker Compose

```yml
version: '3.6'

services:
  postgres:
    container_name: 'nest-clean-pg' # importante definir para não gerar um id aleatório
    image: postgres
    ports:
      - 5432:5432 # mapeia a porta do container para a porta da máquina local
    environment:
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: docker
      POSTGRES_DB: nest-clean
      PGDATA: /data/postgres
    volumes: # mapeia a pasta do host para o container, garantindo que os dados sejam persistidos mesmo que o container seja destruído
      - ./data/pg:/data/postgres
```

## Prisma Service

```ts
import { Injectable, OnModuleDestroy, OnModuleInit } from '@nestjs/common'
import { PrismaClient } from '@prisma/client/scripts/default-index.js'

@Injectable()
export class PrismaService
  extends PrismaClient
  implements OnModuleInit, OnModuleDestroy
{
  constructor() {
    super({
      log: ['error', 'warn'],
    })
  }

  // Métodos que o nest chama automaticamente quando o módulo que usa esse serviço, for instanciado ou destruído. Assim, garantimos que a conexão com o banco de dados seja estabelecida e encerrada corretamente.

  onModuleInit() {
    return this.$connect()
  }

  onModuleDestroy() {
    return this.$disconnect()
  }
}
```
